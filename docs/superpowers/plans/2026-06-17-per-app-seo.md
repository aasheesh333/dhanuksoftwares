# Per-App SEO Pages + AI Auto-fill Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a per-app SEO page system on Netlify where every app has its own indexable HTML page, with AI-powered content generation via Groq (llama-3.3-70b-versatile) and GitHub auto-deploy from the admin panel.

**Architecture:** Static site with build-time generation. Admin form posts to Netlify Functions (Groq proxy, GitHub commit, tracking). Build script (`build.mjs`) reads `apps.json`, renders `/apps/{slug}/index.html` from `template.html` for each app, generates sitemap/robots. Markdown pages convert via template substitution + JSON-LD injection.

**Tech Stack:** Node.js 20 (Netlify build), vanilla HTML/CSS/JS, Netlify Functions, Groq API, GitHub REST API.

// lib/slug.mjs
export function slugify(name) {
  if (!name) return '';
  return String(name)
    .toLowerCase()
    .normalize('NFKD')
    .replace(/[\u0300-\u036f]/g, '')
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/^-+|-+$/g, '')
    .replace(/-{2,}/g, '-');
}

export function ensureUnique(slug, existing) {
  if (!existing.has(slug)) return slug;
  let n = 2;
  while (existing.has(`${slug}-${n}`)) n++;
  return `${slug}-${n}`;
}
```

// lib/migrate.mjs
const MARKETPLACE_MAP = {
  playUrl:     { name: 'Google Play', type: 'play' },
  uptodownUrl: { name: 'Uptodown',    type: 'uptodown' },
  oppoUrl:     { name: 'OPPO Store',  type: 'oppo' },
  vivoUrl:     { name: 'Vivo Store',  type: 'vivo' }
};

export function migrateApp(app) {
  if (!app) return app;
  if (Array.isArray(app.marketplaces)) return app;
  const marketplaces = [];
  for (const [field, meta] of Object.entries(MARKETPLACE_MAP)) {
    const url = (app[field] || '').trim();
    if (url) marketplaces.push({ name: meta.name, url, type: meta.type });
  }
  const result = { ...app };
  delete result.playUrl;
  delete result.uptodownUrl;
  delete result.oppoUrl;
  delete result.vivoUrl;
  if (app.desc && !app.shortDesc) {
    result.shortDesc = app.desc;
    delete result.desc;
  }
  if (!app.tagline && app.shortDesc) {
    result.tagline = app.shortDesc.slice(0, 60);
  }
  if (!app.slug && app.name) {
    result.slug = app.name.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/^-+|-+$/g, '');
  }
  if (!app.longDescription) {
    result.longDescription = result.shortDesc || '';
  }
  if (!app.keywords) result.keywords = [];
  if (!app.features) result.features = [];
  if (!app.faq) result.faq = [];
  if (!app.screenshots) result.screenshots = [];
  if (!app.rating) result.rating = { value: 0, count: 0 };
  result.marketplaces = marketplaces;
  return result;
}

export function migrateAll(apps) {
  if (!Array.isArray(apps)) return [];
  return apps.map(migrateApp);
}
```

// lib/render.mjs
import fs from 'node:fs';
import path from 'node:path';

function escapeHtml(str) {
  if (str == null) return '';
  return String(str)
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#039;');
}

export function slugifyForUrl(name) {
  return String(name || '').toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/^-+|-+$/g, '');
}

export function addUtm(url, slug, marketplace, source) {
  if (!url) return url;
  try {
    const u = new URL(url);
    u.searchParams.set('utm_source', 'dhanuksoftwares');
    u.searchParams.set('utm_medium', 'website');
    u.searchParams.set('utm_campaign', 'app_page');
    u.searchParams.set('utm_content', `${slug}_${source}`);
    u.searchParams.set('utm_term', marketplace);
    return u.toString();
  } catch (e) {
    return url;
  }
}

export function buildSchemaSoftwareApp(app, baseUrl) {
  const offers = (app.marketplaces || []).map(m => ({
    '@type': 'Offer',
    url: m.url,
    availability: 'https://schema.org/InStock',
    price: '0',
    priceCurrency: 'INR'
  }));
  const schema = {
    '@context': 'https://schema.org',
    '@type': 'SoftwareApplication',
    name: app.name,
    description: app.shortDesc,
    applicationCategory: app.category || 'UtilitiesApplication',
    operatingSystem: 'Android',
    offers,
    url: `${baseUrl}/apps/${app.slug}/`
  };
  if (app.icon) schema.image = app.icon;
  if (app.screenshots && app.screenshots[0]) schema.screenshot = app.screenshots[0];
  if (app.rating && app.rating.count > 0) {
    schema.aggregateRating = {
      '@type': 'AggregateRating',
      ratingValue: app.rating.value,
      reviewCount: app.rating.count
    };
  }
  if (app.lastUpdated) schema.datePublished = app.lastUpdated;
  return JSON.stringify(schema, null, 2);
}

export function buildSchemaFAQ(app) {
  const schema = {
    '@context': 'https://schema.org',
    '@type': 'FAQPage',
    mainEntity: (app.faq || []).map(f => ({
      '@type': 'Question',
      name: f.q,
      acceptedAnswer: { '@type': 'Answer', text: f.a }
    }))
  };
  return JSON.stringify(schema, null, 2);
}

export function buildSchemaBreadcrumb(app, baseUrl) {
  const schema = {
    '@context': 'https://schema.org',
    '@type': 'BreadcrumbList',
    itemListElement: [
      { '@type': 'ListItem', position: 1, name: 'Home', item: baseUrl + '/' },
      { '@type': 'ListItem', position: 2, name: 'Apps', item: baseUrl + '/apps/' },
      { '@type': 'ListItem', position: 3, name: app.name, item: `${baseUrl}/apps/${app.slug}/` }
    ]
  };
  return JSON.stringify(schema, null, 2);
}

function renderFeatures(features) {
  if (!features || !features.length) return '';
  return features.map(f => `      <div class="feature-card"><div class="feature-icon">✓</div><div class="feature-title">${escapeHtml(f)}</div></div>`).join('\n');
}

function renderScreenshots(screenshots) {
  if (!screenshots || !screenshots.length) return '';
  return screenshots.map((s, i) => `      <img src="${escapeHtml(s)}" alt="screenshot ${i + 1}" loading="lazy"/>`).join('\n');
}

function renderFAQ(faq) {
  if (!faq || !faq.length) return '';
  return faq.map(f => `    <div class="faq-item"><div class="faq-q">${escapeHtml(f.q)}</div><div class="faq-a">${escapeHtml(f.a)}</div></div>`).join('\n');
}

function renderSecondary(marketplaces, slug) {
  if (!marketplaces || marketplaces.length < 2) return '';
  return marketplaces.slice(1).map(m =>
    `        <a class="btn-secondary" href="${escapeHtml(addUtm(m.url, slug, m.type, 'hero_more'))}" onclick="trackDl('${escapeHtml(slug)}','${escapeHtml(m.type)}','hero_more')" rel="noopener">${escapeHtml(m.name)}</a>`
  ).join('\n');
}

function renderRelated(related) {
  if (!related || !related.length) return '';
  return related.map(r =>
    `      <a class="related-card" href="/apps/${escapeHtml(r.slug)}/"><div class="related-emoji">${escapeHtml(r.emoji || '📱')}</div><div class="related-name">${escapeHtml(r.name)}</div><div class="related-desc">${escapeHtml(r.shortDesc || '')}</div></a>`
  ).join('\n');
}

export function renderApp(app, baseUrl, relatedApps) {
  const marketplaces = app.marketplaces || [];
  const primary = marketplaces[0];

  html = html.replace(/\{\{shortDesc\}\}/g, escapeHtml(app.shortDesc || ''));
  html = html.replace(/\{\{longDescription\}\}/g, escapeHtml(app.longDescription || ''));
  html = html.replace(/\{\{keywordsString\}\}/g, escapeHtml(keywordsString));
  html = html.replace(/\{\{emoji\}\}/g, escapeHtml(app.emoji || '📱'));
  html = html.replace(/\{\{icon\}\}/g, escapeHtml(app.icon || ''));
  html = html.replace(/\{\{category\}\}/g, escapeHtml(app.category || 'Utilities'));
  html = html.replace(/\{\{lastUpdated\}\}/g, escapeHtml(app.lastUpdated || ''));
  html = html.replace(/\{\{primaryScreenshot\}\}/g, escapeHtml(primaryScreenshot));

  // Primary download
  if (primary) {
    html = html.replace(/\{\{primaryDownloadUrl\}\}/g, escapeHtml(addUtm(primary.url, app.slug, primary.type, 'primary')));
    html = html.replace(/\{\{primaryMarketplaceName\}\}/g, escapeHtml(primary.name));
    html = html.replace(/\{\{primaryMarketplaceType\}\}/g, escapeHtml(primary.type));
    html = html.replace(/\{\{#if hasPrimaryDownload\}\}[\s\S]*?\{\{else\}\}[\s\S]*?\{\{\/if\}\}/g, '');
    html = html.replace(/\{\{#if hasPrimaryDownload\}\}/g, '');
    html = html.replace(/\{\{\/if\}\}/g, '');
  } else {
    html = html.replace(/\{\{#if hasPrimaryDownload\}\}[\s\S]*?\{\{\/if\}\}/g, '');
    html = html.replace(/\{\{primaryDownloadUrl\}\}/g, 'mailto:support@dhanuksoftwares.com');
    html = html.replace(/\{\{primaryMarketplaceName\}\}/g, 'Contact Us');
    html = html.replace(/\{\{primaryMarketplaceType\}\}/g, 'contact');
  }

  // Secondary downloads
  if (secondary.length > 0) {
    const secondaryHtml = secondary.map(m =>
      `<a class="btn-secondary" href="${escapeHtml(addUtm(m.url, app.slug, m.type, 'secondary'))}" onclick="trackDl('${escapeHtml(app.slug)}','${escapeHtml(m.type)}','secondary')" rel="noopener">${escapeHtml(m.name)}</a>`
    ).join('\n');
    html = html.replace(/\{\{#each secondaryDownloads\}\}[\s\S]*?\{\{\/each\}\}/g, secondaryHtml);
    html = html.replace(/\{\{#if hasSecondaryDownloads\}\}/g, '');
    html = html.replace(/\{\{secondaryCount\}\}/g, String(secondary.length));
  } else {
    html = html.replace(/\{\{#if hasSecondaryDownloads\}\}[\s\S]*?\{\{\/if\}\}/g, '');
  }

  // Conditional sections
  html = html.replace(/\{\{#if features\.length\}\}([\s\S]*?)\{\{\/if\}\}/g, (m, body) =>
    (app.features && app.features.length) ? body : ''
  );
  html = html.replace(/\{\{#if screenshots\.length\}\}([\s\S]*?)\{\{\/if\}\}/g, (m, body) =>
    (app.screenshots && app.screenshots.length) ? body : ''
  );
  html = html.replace(/\{\{#if faq\.length\}\}([\s\S]*?)\{\{\/if\}\}/g, (m, body) =>
    (app.faq && app.faq.length) ? body : ''
  );
  html = html.replace(/\{\{#if relatedApps\.length\}\}([\s\S]*?)\{\{\/if\}\}/g, (m, body) =>
    (relatedApps && relatedApps.length) ? body : ''
  );
  html = html.replace(/\{\{#if hasRating\}\}([\s\S]*?)\{\{\/if\}\}/g, (m, body) =>
    (app.rating && app.rating.count > 0) ? body : ''
  );
  html = html.replace(/\{\{ratingValue\}\}/g, String(app.rating?.value || 0));
  html = html.replace(/\{\{ratingCount\}\}/g, String(app.rating?.count || 0));

  // Each blocks (features, faq, screenshots, related)
  html = html.replace(/\{\{#each features\}\}([\s\S]*?)\{\{\/each\}\}/g, () => renderFeatures(app.features));
  html = html.replace(/\{\{#each faq\}\}([\s\S]*?)\{\{\/each\}\}/g, () => renderFAQ(app.faq));

  html = html.replace(/\{\{schemaFAQ\}\}/g, buildSchemaFAQ(app));
  html = html.replace(/\{\{schemaBreadcrumb\}\}/g, buildSchemaBreadcrumb(app, baseUrl));

  return html;
}

function rmrf(p) {
  if (fs.existsSync(p)) fs.rmSync(p, { recursive: true, force: true });
}

function mkdirp(p) {
  fs.mkdirSync(p, { recursive: true });
}

function copyFile(src, dest) {
  mkdirp(path.dirname(dest));
  fs.copyFileSync(src, dest);
}

function copyDir(src, dest) {
  if (!fs.existsSync(src)) return;
  mkdirp(dest);
  for (const entry of fs.readdirSync(src, { withFileTypes: true })) {
    const s = path.join(src, entry.name);
    const d = path.join(dest, entry.name);
    if (entry.isDirectory()) copyDir(s, d);
    else copyFile(s, d);
  }
}

function readJson(p) {
  return JSON.parse(fs.readFileSync(p, 'utf8'));
}

function writeFile(p, content) {
  mkdirp(path.dirname(p));
  fs.writeFileSync(p, content);
}

async function main() {
  console.log('🔨 Building dhanuksoftwares site...');

  // 1. Clean dist
  rmrf(DIST);
  mkdirp(DIST);

  // 2. Migrate + load apps
  const rawApps = readJson(path.join(ROOT, 'apps.json'));
  const apps = migrateAll(rawApps);
  console.log(`📦 ${apps.length} apps loaded`);

  // 3. Ensure slugs unique
  const usedSlugs = new Set();
  for (const app of apps) {
    if (!app.slug) app.slug = slugify(app.name);
    app.slug = ensureUnique(app.slug, usedSlugs);
    usedSlugs.add(app.slug);
  }

  // 4. Render per-app pages
  for (const app of apps) {
    const related = apps
      .filter(a => a.slug !== app.slug)
      .slice(0, 4)
      .map(a => ({ name: a.name, slug: a.slug, emoji: a.emoji, shortDesc: a.shortDesc }));
    const html = renderApp(app, BASE_URL, related);
    const outPath = path.join(DIST, 'apps', app.slug, 'index.html');
    writeFile(outPath, html);
    console.log(`  ✓ /apps/${app.slug}/`);
  }

  // 5. Apps index page
  const appsIndexHtml = renderAppsIndex(apps, BASE_URL);
  writeFile(path.join(DIST, 'apps', 'index.html'), appsIndexHtml);
  console.log('  ✓ /apps/');

  // 6. Copy home page
  const homeSrc = path.join(ROOT, 'index.html');
  if (fs.existsSync(homeSrc)) {
    fs.copyFileSync(homeSrc, path.join(DIST, 'index.html'));
    console.log('  ✓ /');
  }

  // 7. Copy admin
  const adminSrc = path.join(ROOT, 'docs', 'admin.html');
  if (fs.existsSync(adminSrc)) {
    const adminDir = path.join(DIST, 'admin');
    mkdirp(adminDir);
    fs.copyFileSync(adminSrc, path.join(adminDir, 'index.html'));
    fs.copyFileSync(adminSrc, path.join(DIST, 'docs', 'admin.html'));
    console.log('  ✓ /admin/');
  }

  // 8. Copy static assets
  for (const f of ['app-ads.txt', 'CNAME', 'robots.txt']) {
    const src = path.join(ROOT, f);
    if (fs.existsSync(src)) {
      fs.copyFileSync(src, path.join(DIST, f));
    }
  }

  // 9. Copy asset directories
  for (const dir of ['icons', 'screens', 'images', 'assets']) {

Return this exact JSON shape:

function checkRate(ip) {
  const now = Date.now();
  const record = rateLimitMap.get(ip) || { count: 0, resetAt: now + RATE_WINDOW };
  if (now > record.resetAt) {
    record.count = 0;
    record.resetAt = now + RATE_WINDOW;
  }
  if (record.count >= RATE_LIMIT) return false;
  record.count++;
  rateLimitMap.set(ip, record);
  return true;
}

  }

  const { name, category, seed } = body;
  if (!name || !seed) {
    return new Response(JSON.stringify({ error: 'Missing name or seed' }), { status: 400, headers });
  }

  const userPrompt = `App name: ${name}
Category: ${category || 'General'}
Short description (seed): ${seed}

Generate the SEO content JSON for this app.`;

    const data = await res.json();
    const content = data.choices?.[0]?.message?.content;
    if (!content) {

    catch (e) {
      return new Response(JSON.stringify({ error: 'Invalid JSON from AI', raw: content.slice(0, 500) }), { status: 502, headers });
    }

    return new Response(JSON.stringify({ ok: true, data: parsed, model: GROQ_MODEL }), { status: 200, headers });
  } catch (e) {
    return new Response(JSON.stringify({ error: `Server error: ${e.message}` }), { status: 500, headers });

  let toastTimer = null;

  function escapeHtml(str) {
    if (str == null) return '';
    return String(str).replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;').replace(/"/g, '&quot;').replace(/'/g, '&#039;');
  }

  function autoSlug(name) {
    if (!name) return '';
    return String(name).toLowerCase().normalize('NFKD').replace(/[\u0300-\u036f]/g, '').replace(/[^a-z0-9]+/g, '-').replace(/^-+|-+$/g, '');
  }

  const MARKETPLACE_TYPES = {
    'Google Play': 'play', 'Uptodown': 'uptodown', 'OPPO Store': 'oppo', 'Vivo Store': 'vivo',
    'Huawei AppGallery': 'huawei', 'Samsung Galaxy Store': 'samsung', 'Amazon Appstore': 'amazon',

  function addMarketplaceRow(name = '', url = '') {
    const list = document.getElementById('marketplaces-list');
    const row = document.createElement('div');
    row.className = 'mp-row';
    row.style.cssText = 'display:grid;grid-template-columns:1fr 2fr auto;gap:0.4rem;margin-bottom:0.4rem;align-items:center;';
    const type = MARKETPLACE_TYPES[name] || 'custom';
    row.innerHTML = `
      <input type="text" class="mp-name" value="${escapeHtml(name)}" placeholder="Marketplace name"/>
      <input type="url" class="mp-url" value="${escapeHtml(url)}" placeholder="https://…"/>
      <button type="button" class="btn btn-danger btn-sm mp-del">✕</button>
    `;
    row.querySelector('.mp-del').addEventListener('click', () => row.remove());
    list.appendChild(row);
  }

  function collectMarketplaces() {
    const rows = document.querySelectorAll('#marketplaces-list .mp-row');
    return Array.from(rows).map(r => {
      const name = r.querySelector('.mp-name').value.trim();
      const url = r.querySelector('.mp-url').value.trim();
      if (!name || !url) return null;
      return { name, url, type: MARKETPLACE_TYPES[name] || 'custom' };
    }).filter(Boolean);
  }

  function loadMarketplaces(marketplaces) {
    const list = document.getElementById('marketplaces-list');
    list.innerHTML = '';
    (marketplaces || []).forEach(mp => addMarketplaceRow(mp.name, mp.url));
    if (!marketplaces || !marketplaces.length) addMarketplaceRow();
  }

  function parseFaq(text) {
    if (!text) return [];
    return text.split('\n').map(l => l.trim()).filter(Boolean).map(line => {
      const m = line.match(/^Q:\s*(.+?)\s*\|\s*A:\s*(.+)$/i);
      if (!m) return null;
      return { q: m[1], a: m[2] };
    }).filter(Boolean);
  }

  function stringifyFaq(faq) {
    if (!Array.isArray(faq)) return '';
    return faq.map(f => `Q: ${f.q} | A: ${f.a}`).join('\n');
  }

  function parseList(text) {
    if (!text) return [];
    return text.split('\n').map(l => l.trim()).filter(Boolean);
  }

  function getFormData() {
    return {

      tag: document.getElementById('f-cat').value,

      longDescription: document.getElementById('f-longDesc').value.trim(),
      keywords: document.getElementById('f-keywords').value.split(',').map(k => k.trim()).filter(Boolean),
      features: parseList(document.getElementById('f-features').value),
      faq: parseFaq(document.getElementById('f-faq').value),
      emoji: document.getElementById('f-emoji').value.trim(),
      icon: document.getElementById('f-icon').value.trim(),
      screenshots: parseList(document.getElementById('f-screenshots').value),
      marketplaces: collectMarketplaces(),
      rating: {
        value: parseFloat(document.getElementById('f-rating-value').value) || 0,
        count: parseInt(document.getElementById('f-rating-count').value) || 0
      },
      lastUpdated: document.getElementById('f-updated').value

  }

  function setFormData(app) {
    document.getElementById('f-name').value = app.name || '';
    document.getElementById('f-slug').value = app.slug || autoSlug(app.name || '');
    document.getElementById('f-cat').value = app.category || app.tag || 'Other';
    document.getElementById('f-tagline').value = app.tagline || '';
    document.getElementById('f-seed').value = '';
    document.getElementById('f-shortDesc').value = app.shortDesc || '';
    document.getElementById('f-longDesc').value = app.longDescription || '';
    document.getElementById('f-keywords').value = (app.keywords || []).join(', ');
    document.getElementById('f-features').value = (app.features || []).join('\n');
    document.getElementById('f-faq').value = stringifyFaq(app.faq);
    document.getElementById('f-emoji').value = app.emoji || '';
    document.getElementById('f-icon').value = app.icon || '';
    document.getElementById('f-screenshots').value = (app.screenshots || []).join('\n');
    loadMarketplaces(app.marketplaces);
    document.getElementById('f-rating-value').value = app.rating?.value || 0;
    document.getElementById('f-rating-count').value = app.rating?.count || 0;
    document.getElementById('f-updated').value = app.lastUpdated || new Date().toISOString().slice(0, 10);
    updateCharCounts();
  }

  function updateCharCounts() {

    document.getElementById('f-slug').dataset.touched = '1';

  // AI fill
  document.getElementById('ai-fill-btn').addEventListener('click', async () => {
    if (aiBusy) return;
    const name = document.getElementById('f-name').value.trim();
    const category = document.getElementById('f-cat').value;
    const seed = document.getElementById('f-seed').value.trim();
    if (!name || !seed || seed.split(/\s+/).length < 3) {
      toast('Please enter App Name and a 3+ word AI seed description first.', 'error');
      return;
    }
    aiBusy = true;
    const btn = document.getElementById('ai-fill-btn');
    btn.disabled = true;
    btn.textContent = '⏳ Generating...';
    try {
      const res = await fetch(AI_FILL_URL, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'x-admin-token': ADMIN_PASSWORD },
        body: JSON.stringify({ name, category, seed })
      });
      const data = await res.json();
      if (!res.ok || !data.ok) {
        throw new Error(data.error || 'AI fill failed');
      }
      const d = data.data;
      if (d.tagline) document.getElementById('f-tagline').value = d.tagline;
      if (d.shortDesc) document.getElementById('f-shortDesc').value = d.shortDesc;
      if (d.longDescription) document.getElementById('f-longDesc').value = d.longDescription;
      if (Array.isArray(d.keywords)) document.getElementById('f-keywords').value = d.keywords.join(', ');
      if (Array.isArray(d.features)) document.getElementById('f-features').value = d.features.join('\n');
      if (Array.isArray(d.faq)) document.getElementById('f-faq').value = stringifyFaq(d.faq);
      updateCharCounts();
      toast('✨ AI content generated. Review and edit before saving.', 'success');
    } catch (e) {
      toast(`AI error: ${e.message}`, 'error');
    } finally {
      aiBusy = false;
      btn.disabled = false;
      btn.textContent = '✨ AI Auto-fill All (Groq)';
    }
  });

  document.getElementById('app-form').addEventListener('submit', async (e) => {
    e.preventDefault();
    const data = getFormData();
    if (!data.name) { toast('App name is required', 'error'); return; }
    if (!data.shortDesc) { toast('Short description is required', 'error'); return; }
    if (!data.longDescription) { toast('Long description is required (or use AI fill)', 'error'); return; }

    const editIdx = parseInt(document.getElementById('edit-index').value, 10);
    if (editIdx >= 0) {
      apps[editIdx] = { ...apps[editIdx], ...data };
    } else {
      // Check slug uniqueness
      if (apps.some(a => a.slug === data.slug)) {
        toast(`Slug "${data.slug}" already exists. Use a different name.`, 'error');
        return;
      }
      apps.push(data);
    }
    renderAppsList();

    const btn = document.getElementById('save-btn');
    btn.disabled = true;
    btn.textContent = '⏳ Publishing...';
    try {
      const res = await fetch(PUBLISH_URL, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'x-admin-token': ADMIN_PASSWORD },
        body: JSON.stringify({ apps })
      });
      const result = await res.json();
      if (!res.ok || !result.ok) {
        throw new Error(result.error || 'Publish failed');
      }
      toast(`✅ Deployed! ${apps.length} apps live. Build will start in ~30s.`, 'success');
      document.getElementById('edit-index').value = '-1';
      document.getElementById('form-title').textContent = 'Add New App';
      btn.textContent = 'Save & Deploy';
    } catch (e) {
      toast(`Publish error: ${e.message}`, 'error');
      // Revert local change
      if (editIdx >= 0) {
        // cannot revert easily
      } else {
        apps.pop();
        renderAppsList();
      }
    } finally {
      btn.disabled = false;
    }
  });

  function renderAppsList() {
    const list = document.getElementById('apps-list');
    const counter = document.getElementById('apps-count');
    counter.textContent = apps.length ? `(${apps.length})` : '';
    if (!apps.length) {
      list.innerHTML = '<div class="empty-msg">No apps yet. Add one using the form.</div>';
      return;
    }
    list.innerHTML = '';
    apps.forEach((app, idx) => {
      const card = document.createElement('div');
      card.className = 'admin-app-card';
      const icon = app.icon
        ? `<img src="${escapeHtml(app.icon)}" alt="" style="width:36px;height:36px;border-radius:8px;object-fit:cover;flex-shrink:0;" onerror="this.outerHTML='<div class=&quot;admin-app-emoji&quot;>${escapeHtml(app.emoji || "📱")}</div>'"/>`
        : `<div class="admin-app-emoji">${escapeHtml(app.emoji || '📱')}</div>`;
      card.innerHTML = `
        ${icon}
        <div class="admin-app-info">
          <div class="admin-app-name">${escapeHtml(app.name || '(unnamed)')}</div>
          <span class="admin-app-cat">${escapeHtml(app.category || app.tag || 'Other')}</span>
        </div>
        <div class="admin-app-actions">
          <a class="btn btn-edit btn-sm" href="/apps/${escapeHtml(app.slug || '')}/" target="_blank" rel="noopener">View</a>
          <button class="btn btn-edit btn-sm" data-action="edit" data-index="${idx}">✎ Edit</button>
          <button class="btn btn-danger btn-sm" data-action="delete" data-index="${idx}">✕</button>
        </div>
      `;
      list.appendChild(card);
    });
    list.querySelectorAll('button[data-action="edit"]').forEach(btn => {
      btn.addEventListener('click', () => {
        const idx = parseInt(btn.dataset.index, 10);
        setFormData(apps[idx]);
        document.getElementById('edit-index').value = String(idx);
        document.getElementById('form-title').textContent = `Edit: ${apps[idx].name}`;
        document.getElementById('save-btn').textContent = 'Update & Deploy';
        window.scrollTo({ top: 0, behavior: 'smooth' });
      });
    });
    list.querySelectorAll('button[data-action="delete"]').forEach(btn => {
      btn.addEventListener('click', () => {
        const idx = parseInt(btn.dataset.index, 10);
        if (confirm(`Delete "${apps[idx].name}"? You still need to click Save & Deploy to publish.`)) {
          apps.splice(idx, 1);
          renderAppsList();
          toast('App removed locally. Click "Save & Deploy" to publish.', 'success');
        }
      });
    });
  }

  async function initPanel() {
    try {
      const res = await fetch(JSON_URL, { cache: 'no-store' });
      if (!res.ok) throw new Error('HTTP ' + res.status);
      const data = await res.json();
      apps = Array.isArray(data) ? data : [];
    } catch (e) {
      apps = [];
      toast('⚠️ Could not load apps.json. Starting fresh.', 'error');
    }
    loadMarketplaces([]);
    renderAppsList();
    clearForm();
    document.getElementById('f-updated').value = new Date().toISOString().slice(0, 10);
  }
